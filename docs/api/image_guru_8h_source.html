42 static ImageType GetImageType(const std::vector<uint8_t> &data) {
43 if (data.empty()) {
44 return ImageType::UNKNOWN;
45 }
46
47 std::array<uint8_t, 12> buffer{};
48 size_t bytes_to_copy = std::min(data.size(), buffer.size());
49 std::copy_n(data.begin(), bytes_to_copy, buffer.begin());
50
51 // Check for JPEG (FF D8)
52 if (data.size() >= 2 && buffer[0] == 0xFF && buffer[1] == 0xD8) {
53 return ImageType::JPEG;
54 }
55
56 // Check for BMP (BM)
57 if (data.size() >= 2 && buffer[0] == 'B' && buffer[1] == 'M') {
58 return ImageType::BMP;
59 }
60
61 // Check for GIF (GIF87a or GIF89a)
62 if (data.size() >= 6 && buffer[0] == 'G' && buffer[1] == 'I' &&
63 buffer[2] == 'F' && buffer[3] == '8' &&
64 (buffer[4] == '7' || buffer[4] == '9') && buffer[5] == 'a') {
65 return ImageType::GIF;
66 }
67
68 // Check for TIFF (II or MM header)
69 if (data.size() >= 4) {
70 bool isTiffLE = (buffer[0] == 'I' && buffer[1] == 'I' &&
71 buffer[2] == 0x2A && buffer[3] == 0x00);
72 bool isTiffBE = (buffer[0] == 'M' && buffer[1] == 'M' &&
73 buffer[2] == 0x00 && buffer[3] == 0x2A);
74 if (isTiffLE || isTiffBE)
75 return ImageType::TIFF;
76 }
77
78 // Check for PNG
79 if (data.size() >= 8) {
80 const std::array<uint8_t, 8> pngSig = {0x89, 'P', 'N', 'G',
81 0x0D, 0x0A, 0x1A, 0x0A};
82 if (std::equal(pngSig.begin(), pngSig.end(), buffer.begin())) {
83 return ImageType::PNG;
84 }
85 }
86
87 // Check for WEBP (RIFF....WEBP)
88 if (data.size() >= 12 && buffer[0] == 'R' && buffer[1] == 'I' &&
89 buffer[2] == 'F' && buffer[3] == 'F' && buffer[8] == 'W' &&
90 buffer[9] == 'E' && buffer[10] == 'B' && buffer[11] == 'P') {
91 return ImageType::WEBP;
92 }
93
94 // Check for AVIF (ftypavif)
95 if (data.size() >= 12 && buffer[4] == 'f' && buffer[5] == 't' &&
96 buffer[6] == 'y' && buffer[7] == 'p' && buffer[8] == 'a' &&
97 buffer[9] == 'v' && buffer[10] == 'i' && buffer[11] == 'f') {
98 return ImageType::AVIF;
99 }
100
101 return ImageType::UNKNOWN;
102 }
