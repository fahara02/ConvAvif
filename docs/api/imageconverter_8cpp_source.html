51 if (width <= 0 || height <= 0 || width > 8192 || height > 8192) {

56 // Load image using STB library
57 int w, h, channels;

59 input_data.data(), input_data.size(), &w, &h, &channels, STBI_rgb_alpha);
60 if (!data)
61 return Error(ConverterError::IMAGE_LOAD_FAILED, "STB image load failed",
62 __func__);
63
64 // Resize image if dimensions don't match requested size
65 std::vector<unsigned char> processed_data;
66 if (w == width && h == height) {
67 processed_data.assign(data, data + (w * h * 4));
68 } else {
69 processed_data.resize(width * height * 4);

72 }

74 stbi_image_free(data);

81 "Failed to create AVIF encoder", __func__);
82 }
83

88 break;
89 case CodecChoice::SVT:

91 break;
92 default:
93 encoder->codecChoice = AVIF_CODEC_CHOICE_AUTO;
94 break;
95 }

99 case Tune::TUNE_PSNR: {
100 Result optionResult =
101 SetAvifOption(encoder.get(), "tune", "psnr", "PSNR Tuning", __func__);
102 if (auto *error = std::get_if<Error>(&optionResult)) {
103 return *error;
104 }
105 break;
106 }
107 case Tune::TUNE_SSIM: {
108 Result optionResult =
109 SetAvifOption(encoder.get(), "tune", "ssim", "SSIM Tuning", __func__);
110 if (auto *error = std::get_if<Error>(&optionResult)) {
111 return *error;
112 }
113 break;
114 }
115 default:
116 break;
117 }
118
119 // Configure general encoder settings
120 encoder->maxThreads = emscripten_num_logical_cores();
121 encoder->quality = config.quality;
122 encoder->qualityAlpha =
123 (config.qualityAlpha == -1) ? config.quality : config.qualityAlpha;
124 encoder->speed = config.getSpeed();
125
126 // Set sharpness option
127 Result sharpnessResult = SetAvifOption(
128 encoder.get(), "sharpness", std::to_string(config.sharpness).c_str(),

131 return *error;

138 int maxQ = (config.maxQuantizer == -1) ? q : config.maxQuantizer;

183 image->transferCharacteristics = AVIF_TRANSFER_CHARACTERISTICS_SRGB;
184 image->matrixCoefficients = AVIF_MATRIX_COEFFICIENTS_BT709;
185 image->yuvRange = AVIF_RANGE_FULL;

187 // Set up RGB image for conversion to YUV
188 avifRGBImage rgb;
189 avifRGBImageSetDefaults(&rgb, image.get());
190 rgb.chromaUpsampling = AVIF_CHROMA_UPSAMPLING_AUTOMATIC;
191 rgb.format = AVIF_RGB_FORMAT_RGBA;
192 rgb.depth = RGB_DEPTH;
193 rgb.pixels = processed_data.data();
194 rgb.rowBytes = width * avifRGBImagePixelSize(&rgb);
195 rgb.alphaPremultiplied = AVIF_FALSE;
196 rgb.ignoreAlpha = false;
197
198 // Convert RGB to YUV
199 avifResult convertResult = avifImageRGBToYUV(image.get(), &rgb);
200 if (convertResult != AVIF_RESULT_OK) {
201 return Error(ConverterError::CONVERSION_FAILED,
202 "RGB->YUV conversion failed: " +
203 std::string(avifResultToString(convertResult)),
204 __func__);
205 }
206
207 // Encode the image to AVIF format
208 avifRWData output = AVIF_DATA_EMPTY;
209 avifResult result = avifEncoderWrite(encoder.get(), image.get(), &output);
210 if (result != AVIF_RESULT_OK) {
211 std::string errorMsg =
212 "Encoding failed: " + std::string(avifResultToString(result));
213 if (result == AVIF_RESULT_OUT_OF_MEMORY) {
214 errorMsg += " - Try higher speed values (5-10)";
215 }
216 return Error(avifToConverterError(result), errorMsg, __func__);
217 }
218
219 // Copy encoded data to result vector
220 std::vector<uint8_t> output_data(output.data, output.data + output.size);
221 avifRWDataFree(&output);
222
223 return std::make_shared<ImageBuffer>(std::move(output_data));
